event ObjectUpdate
	// change every Ring entity in the stage into a Blank Object
	foreach (TypeName[Ring], arrayPos0, ALL_ENTITIES)
		object[arrayPos0].type = TypeName[Blank Object]
	next
	
	object.type = TypeName[Blank Object]
end event


event ObjectStartup
	foreach (TypeName[NoRingMode], arrayPos0, ALL_ENTITIES)
		object[arrayPos0].priority = PRIORITY_ACTIVE
	next
end event
